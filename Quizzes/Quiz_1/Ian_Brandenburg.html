# loading and processing the data
def load_and_preprocess_document(url):
    
    # fetch the content of the file from github
    response = requests.get(url)
    content = response.text

    # split the speaches into documents by <DOC> using regex
    speeches = re.findall('<DOC>.*?</DOC>', content, re.DOTALL)
    preprocessed_speeches = []

    # for loop to preprocess the text inside the speaches
    for speech in speeches:
        # extract text within <TEXT> tags using regex
        text = re.search('<TEXT>(.*?)</TEXT>', speech, re.DOTALL).group(1)
        # preprocess the text (lowercase, remove stopwords, tokenize)
        tokens = word_tokenize(text.lower())
        
        # Remove normal stopwords and numbers
        filtered_tokens = [word for word in tokens if word.isalpha() and word not in stopwords.words('english')]
        
        # Remove punctuation
        filtered_tokens = [word for word in filtered_tokens if word not in string.punctuation and not any(char in set(string.punctuation) for char in word)]
        
        preprocessed_speeches.append(' '.join(filtered_tokens))

    return preprocessed_speeches

def apply_stemming(preprocessed_speeches):
    stemmer = PorterStemmer()
    stemmed_speeches = []
    for speech in preprocessed_speeches:
        tokens = speech.split()  # Assuming speech is a string of tokens
        stemmed_tokens = [stemmer.stem(token) for token in tokens]
        stemmed_speeches.append(' '.join(stemmed_tokens))
    return stemmed_speeches

def apply_lemmatization(preprocessed_speeches):
    lemmatizer = WordNetLemmatizer()
    lemmatized_speeches = []
    for speech in preprocessed_speeches:
        tokens = speech.split()
        lemmatized_tokens = [lemmatizer.lemmatize(token) for token in tokens]
        lemmatized_speeches.append(' '.join(lemmatized_tokens))
    return lemmatized_speeches

def apply_droplist(preprocessed_speeches, droplist):
    filtered_speeches = []
    for speech in preprocessed_speeches:
        tokens = speech.split()
        filtered_tokens = [token for token in tokens if token not in droplist]
        filtered_speeches.append(' '.join(filtered_tokens))
    return filtered_speeches

# Initializing a list to store the results
cosine_similarity_results = []

def cosine_similarity_plot(tfidf_matrix, title):
    global cosine_similarity_results
    
    # Step 1: Calculating Cosine Similarity
    cosine_similarities = cosine_similarity(tfidf_matrix[0:1], tfidf_matrix).flatten()

    # Step 2: Finding the Most Similar Speech
    # Ignore the first speech itself by setting its similarity to -1
    cosine_similarities[0] = -1
    most_similar_index = np.argmax(cosine_similarities)
    similarity_score = cosine_similarities[most_similar_index]

    print('_'*110)
    print(f"The most similar speech to the first one is at index {most_similar_index} with a cosine similarity score of {similarity_score}.")
    print('_'*110)

    # Appending results to the global list
    cosine_similarity_results.append({
        'Title': title,
        'Most Similar Index': most_similar_index,
        'Similarity Score': similarity_score
    })

    # Step 3: Analyzing Similarity Over Time
    # Plotting the similarity of each speech to the first speech over time
    plt.plot(cosine_similarities[1:], color='purple') # [1:] skips the first speech
    plt.title(f"Similarity of Each Speech to the First Speech Over Time: {title}")
    plt.xlabel("Speech Index")
    plt.ylabel("Cosine Similarity Score")
    filename = f'cosine_plot_{title.replace(" ", "_")}.png'
    plt.savefig('Plots/' + filename)
    plt.show()

Find the most similar speech to the very first speech for the senator you choose. Are the speeches becoming more similar or dissimilar over time? Explore different measures of similarity.¶

The speeches for senator Inouye of Hawaii were analzyed using cosine similarity. The text underwent six different types of processing. Every process included removing the NLTK stop words, making the text lowercase, removing punctuation and symbols, as well as removing numbers.

There were functions developed to streamline the code instead of having to copy and paste the same code too many times. The first code was the preprocessing text function, which also read the speeches in, dividing them by the DOC HTML code, and extracting the text from between the TEXT HTML code using regular expressions. This list of speeches was fed through a for-loop inside the function, which extracted the stop words, punctuation, symbols, and numbers. The next three functions were constructed to be run on the preprocessed text for similarity comparisons. This includes a stemming, lemmatization, and additionally removing the list of drop words from the dropword.txt file from the preprocessed text. This resulted in six similarity scores for each process to determine which process was the most effective, and also to compare the index value of most similar index to the first (index[0]) speech.

Title Most Similar Index Similarity Score
W/o Lem/Stem or Droplist 147 0.233828
W/o Lem/Stem, w/ Droplist 147 0.233828
W/ Stem, w/o Droplist 147 0.264769
W/ Stem and Droplist 147 0.278640
W/ Lem, W/o Droplist 147 0.242857
W/ Lem and Droplist 147 0.251337

The table above shows that the most similar speech to the first speech is always 147, no matter the method. Each of the similarity scores are different from each other, with the exception of the first two. When the drop list is applied to the non-stemming/lemmatized processed text, there is no difference in similarity scores. Interestingly, there is a consistently a difference when stemming and lemmatization is applied.

The process with the highest similarity score is the stemming process with the droplist removed from the speeches, while the lowest is the first two methods. This could suggest that the droplist, stemming, and lemmatization process effectively remove unnecessary data from the text in order to create better comparisons.

Finally, when reflecting on the time series graph, the speeches stay consistently close similarity scores to around the 0.05 mark. This is likely because most of the speeches are not extremely similar, but have compontents that repeat throughout. There are a few spikes throughout the graph that illustrate potentially similar subjects arrising throughout the politians career, as political issues tend to resurface over the years or throughout a single year. This can be seen in the plot above.
